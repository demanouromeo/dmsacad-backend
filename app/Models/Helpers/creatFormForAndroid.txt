SizedBox _createFormAndroid(BuildContext context) {

      child: Padding(
        padding: const EdgeInsets.symmetric(vertical: 20),
        child: Form(
          autovalidateMode: AutovalidateMode.onUserInteraction,
          key: formKey,
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              TextFormField(
                controller: controlerName,
                //initialValue: specialityName,
                decoration: InputDecoration(
                  border: OutlineInputBorder(
                    borderRadius: BorderRadius.circular(gRaduisCircularVal),
                  ),
                  isDense: true,
                  contentPadding: const EdgeInsets.symmetric(
                    vertical: 20,
                    horizontal: 30,
                  ),
                  hintText:
                      AppLocalizations.of(context)!.speciality, //'Speciality',
                  //prefixIcon: Icon(Icons.lock_rounded),
                  prefixIcon: const Icon(Icons.input),
                  //++++ password
                  labelText: AppLocalizations.of(context)!.speciality,
                  suffixIcon: IconButton(
                    icon: const Icon(
                      Icons.cancel,
                      //size: 20,
                    ),
                    onPressed: () => setState(
                      () {
                        classeName = "";
                        //debugPrint("Speciality: [$specialityName]");
                        controlerName.clear();
                        //controlerName.text = "";
                      },
                    ),
                  ),
                ),
                onChanged: (str) => setState(() => classeName = str),
                validator: (value2) => value2 != null &&
                        (value2.isNotEmpty &&
                            !validateString(value2, allowedCharacters3))
                    ? AppLocalizations.of(context)!.wrongInput //'Wrong input'

                height: 15,

                controller: controlerDescription,

                    borderRadius: BorderRadius.circular(gRaduisCircularVal),

                    horizontal: 30,

                  //++++ password

                  //++++ password

                    icon: const Icon(
                      Icons.cancel,
                      //size: 20,

                      () {
                        description = "";
                        //debugPrint("Speciality: [$description]");
                        controlerDescription.clear();
                      },
                    ),
                  ),

                //controller: passwordController,
                onChanged: (str) => setState(() => description = str),
                validator: (value2) {
                  return null;
                  //Validation here

                height: 15,
              ),
              DropdownButtonFormField2<String>(

                decoration: InputDecoration(
                  contentPadding:
                      const EdgeInsets.symmetric(vertical: 18, horizontal: 10),

                    borderRadius: BorderRadius.circular(gRaduisCircularVal),
                  ),

                          style: const TextStyle(

                          ),

                      ),
                    )

                children: [

                      onPressed: () {
                        String spName = controlerName.text;
                        String description = controlerDescription.text;
                        if (spName.isEmpty) {
                          showSnackBar1(
                              AppLocalizations.of(context)!.infoTitle,
                              AppLocalizations.of(context)!.wrongInput,
                              1,

                              context);
                        } else {
                          bool res = validateString(spName, allowedCharacters3);
                          if (res) {
                            saveSpeciality();

                    height: 45,
                    width: 115,

                        controlerDescription.clear();
                        controlerName.clear();
                        //controlerName.text=" ";
                      },
                      icon: const Icon(
                        Icons.clear,
                        size: 15,
                      ),
                      label: Text(
                        AppLocalizations.of(context)!.clearAll,
                        style: TextStyle(
                          fontSize: 11,
                        ),
                      ),
                    ),
                  ),
                  SizedBox(
                    height: 45,
                    width: 115,
                    child: ElevatedButton.icon(
                      onPressed: () {
                        refreshSpTable();
                      },
                      icon: const Icon(
                        Icons.refresh,
                        size: 15,
                      ),
                      label: Text(
                        AppLocalizations.of(context)!.refresh,
                        style: TextStyle(
                          fontSize: 11,
                        ),
                      ),
                    ),
                  ),
                ],
              )
            ],
          ),
        ),
      ),
